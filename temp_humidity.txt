#include "DHT.h"

#define DHTPIN 2        // Data pin connected to Arduino pin 2
#define DHTTYPE DHT22   // Use DHT22 instead of DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
  Serial.println("Weather Station Initialized (DHT22)");
}

void loop() {
  float temp = dht.readTemperature();      // Celsius
  float hum = dht.readHumidity();          // Percentage

  if (isnan(temp) || isnan(hum)) {
    Serial.println("Failed to read from DHT22 sensor!");
    return;
  }

  Serial.print("Temperature: ");
  Serial.print(temp);
  Serial.println(" Â°C");

  Serial.print("Humidity: ");
  Serial.print(hum);
  Serial.println(" %");

  Serial.println("----------------------");
  delay(2000);
}
